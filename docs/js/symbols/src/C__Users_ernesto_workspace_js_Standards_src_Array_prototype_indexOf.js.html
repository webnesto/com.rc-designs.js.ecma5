  6 if(!Array.prototype.indexOf){ //IE doesn't support indexOf
  7 	Array.prototype.indexOf = function (elt /*, from*/)
  8 	{
  9 	  var len = this.length;
 10 	
 11 	  var from = Number(arguments[1]) || 0;
 12 	  from = (from < 0)
 13 	       ? Math.ceil(from)
 14 	       : Math.floor(from);
 15 	  if (from < 0)
 16 	    from += len;
 17 	
 18 	  for (; from < len; from++)
 19 	  {
 20 	    if (from in this &&
 21 	        this[from] === elt)
 22 	      return from;
 23 	  }
 24 	  return -1;
 25 	};
